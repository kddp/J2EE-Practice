
create table users(sl_no int generated by default as identity primary key, 
email varchar(200) unique, pass varchar(200) not null, role varchar(200)) 

create table userdetails(user_sl int references users(sl_no), fname varchar(200),
lname varchar(200), gender char(1), dob date, cr_dt date)

create table profile(user_sl int references users(sl_no), moreme varchar(200), 
pic blob)
